def start_shopping():
    grand_total = 0  # running total
    while True:
        print("\nAvailable Categories:")
        cat = ["1. Women", "2. Men", "3. Kids"]
        for c in cat:

            choice = int(input("\nSelect category (1-3): "))
            grand_total += clothing(choice)
        except ValueError:

        print(f"\n🧾 Current Total Bill: ₹{grand_total}")

        # Continue shopping or exit
        again = input("\nDo you want to continue shopping? (yes/no): ").lower()

            print(f"Grand Total Amount: ₹{grand_total}")
